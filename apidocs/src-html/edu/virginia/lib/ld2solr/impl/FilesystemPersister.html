001package edu.virginia.lib.ld2solr.impl;
002
003import static org.slf4j.LoggerFactory.getLogger;
004
005import java.io.File;
006import java.io.IOException;
007import java.io.UnsupportedEncodingException;
008import java.net.URLEncoder;
009
010import org.slf4j.Logger;
011
012import com.google.common.io.Files;
013
014import edu.virginia.lib.ld2solr.api.OutputRecord;
015import edu.virginia.lib.ld2solr.spi.ThreadedStage;
016import edu.virginia.lib.ld2solr.spi.RecordSink.RecordPersister;
017
018/**
019 * A {@link RecordPersister} that writes {@link OutputRecord}s to a filesystem.
020 * 
021 * @author ajs6f
022 * 
023 */
024public class FilesystemPersister extends ThreadedStage<FilesystemPersister, OutputRecord> implements RecordPersister {
025
026        private File directory;
027
028        private static final Logger log = getLogger(FilesystemPersister.class);
029
030        /*
031         * (non-Javadoc)
032         * 
033         * @see
034         * edu.virginia.lib.ld2solr.spi.RecordSink#accept(edu.virginia.lib.ld2solr
035         * .api.OutputRecord)
036         */
037        @Override
038        public void accept(final OutputRecord record) {
039                final String fileName;
040                try {
041                        fileName = URLEncoder.encode(record.id(), "UTF-8");
042                } catch (final UnsupportedEncodingException e) {
043                        throw new AssertionError();
044                }
045                final File file = new File(directory, fileName);
046                log.debug("Persisting to: {}", file.toString());
047                try {
048                        Files.asByteSink(file).write(record.record());
049                        log.debug("Persisted: {} to: {}", record.id(), fileName);
050                } catch (final IOException e) {
051                        log.error("Error persisting: {}!", record.id());
052                        log.error("Exception: ", e);
053                }
054        }
055
056        /*
057         * (non-Javadoc)
058         * 
059         * @see
060         * edu.virginia.lib.ld2solr.spi.RecordSink.RecordPersister#location(java
061         * .lang.String)
062         */
063        @Override
064        public FilesystemPersister location(final String location) {
065                directory = new File(location);
066                return this;
067        }
068
069        /*
070         * (non-Javadoc)
071         * 
072         * @see edu.virginia.lib.ld2solr.spi.RecordSink.RecordPersister#location()
073         */
074        @Override
075        public String location() {
076                return directory.getAbsolutePath();
077        }
078
079        @Override
080        public void andThen(final Acceptor<OutputRecord, ?> a) {
081                throw new UnsupportedOperationException(
082                                "It is currently not expected that indexing workflow will continue after persistence.");
083        }
084}
